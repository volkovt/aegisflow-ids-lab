# 1) Feche o VirtualBox GUI se estiver aberto.
#    Depois, mate qualquer processo remanescente (VBoxSVC, VBoxHeadless, VirtualBoxVM)
$procs = "VirtualBoxVM","VBoxHeadless","VBoxSVC","VirtualBox"
foreach ($p in $procs) {
  Get-Process $p -ErrorAction SilentlyContinue | ForEach-Object {
    Write-Host "[Kill] $($_.ProcessName) pid=$($_.Id)"
    Stop-Process -Id $_.Id -Force
  }
}

# 2) Descubra a VM e o UUID
$vmNameHint = "lab_sensor"   # ajuste se necessário
$vb = "C:\Program Files\Oracle\VirtualBox\VBoxManage.exe"
& $vb list vms | Select-String $vmNameHint | ForEach-Object { $_.Line }

# Se aparecer algo como:  "lab_sensor_1759618808905_46127" {966aaf2b-06a1-4581-b822-951baa27d701}
# copie o UUID (entre {}):
$uuid = "966aaf2b-06a1-4581-b822-951baa27d701"  # <- cole aqui

# 3) Caso a VM esteja (ou ache que esteja) rodando/saved, force poweroff e descarte estado
try {
  & $vb list runningvms | Select-String $uuid | Out-Null
  if ($LASTEXITCODE -eq 0) {
    Write-Host "[VBox] Poweroff..."
    & $vb controlvm $uuid poweroff
  }
} catch {}

Write-Host "[VBox] Discard saved state (ignora erro se não houver)"
& $vb discardstate $uuid 2>$null

# 4) Tente UNREGISTER/DELETE diretamente
Write-Host "[VBox] unregistervm --delete"
& $vb unregistervm $uuid --delete

# 5) Limpeza de estados do Vagrant (stale entries)
Write-Host "[Vagrant] global-status --prune"
vagrant global-status --prune

# 6) Volte à pasta do seu lab e destrua pelo nome, por via das dúvidas
